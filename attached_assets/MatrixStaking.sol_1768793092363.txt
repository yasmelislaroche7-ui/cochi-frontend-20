pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
import "@openzeppelin/contracts/utils/ReentrancyGuard.sol";

contract MatrixStaking is ReentrancyGuard {
    using SafeERC20 for IERC20;

    IERC20 public immutable token;
    address public immutable owner;

    uint256 public apr; // 500 = 500%
    uint256 public constant YEAR = 365 days;
    uint256 public constant UNSTAKE_DELAY = 1 days;
    uint256 public constant MAX_APR = 1000; // 1000%

    struct StakeInfo {
        uint256 amount;
        uint256 rewardDebt;
        uint256 lastUpdate;
        uint256 unlockTime;
    }

    mapping(address => StakeInfo) public stakes;

    uint256 public totalStaked;
    uint256 public totalRewardsPaid;

    event Staked(address indexed user, uint256 amount);
    event Unstaked(address indexed user, uint256 amount);
    event RewardsClaimed(address indexed user, uint256 reward);
    event APRUpdated(uint256 newApr);
    event RewardsFunded(uint256 amount);
    event EmergencyWithdraw(address indexed owner, uint256 amount);
    event ERC20Recovered(address token, uint256 amount);

    modifier onlyOwner() {
        require(msg.sender == owner, "NOT_OWNER");
        _;
    }

    constructor(address _token, uint256 _apr) {
        require(_token != address(0), "TOKEN_ZERO");
        require(_apr > 0 && _apr <= MAX_APR, "INVALID_APR");
        
        token = IERC20(_token);
        apr = _apr;
        owner = msg.sender;
    }

    /* ================= STAKING ================= */
    function stake(uint256 amount) external nonReentrant {
        require(amount > 0, "AMOUNT_ZERO");

        StakeInfo storage s = stakes[msg.sender];

        if (s.amount > 0) {
            _updateRewards(msg.sender);
        } else {
            s.lastUpdate = block.timestamp;
        }

        token.safeTransferFrom(msg.sender, address(this), amount);

        s.amount += amount;
        s.unlockTime = block.timestamp + UNSTAKE_DELAY;
        totalStaked += amount;

        emit Staked(msg.sender, amount);
    }

    function unstake(uint256 amount) external nonReentrant {
        require(amount > 0, "AMOUNT_ZERO");

        StakeInfo storage s = stakes[msg.sender];
        require(block.timestamp >= s.unlockTime, "LOCKED");
        require(s.amount >= amount, "INSUFFICIENT");

        _updateRewards(msg.sender);
        _claimRewards(msg.sender);

        s.amount -= amount;
        totalStaked -= amount;

        // ðŸ”’ Reset lock si aÃºn queda stake
        if (s.amount > 0) {
            s.unlockTime = block.timestamp + UNSTAKE_DELAY;
        } else {
            s.lastUpdate = 0; // Cleanup
        }

        token.safeTransfer(msg.sender, amount);
        emit Unstaked(msg.sender, amount);
    }

    function claim() external nonReentrant {
        _updateRewards(msg.sender);
        _claimRewards(msg.sender);
    }

    /* ================= INTERNAL ================= */
    function _updateRewards(address user) internal {
        StakeInfo storage s = stakes[user];

        if (s.amount > 0 && s.lastUpdate > 0) {
            uint256 pending = _calculateNewRewards(user);
            if (pending > 0) {
                s.rewardDebt += pending;
            }
        }
        s.lastUpdate = block.timestamp;
    }

    function _calculateNewRewards(address user) internal view returns (uint256) {
        StakeInfo storage s = stakes[user];
        if (s.amount == 0 || s.lastUpdate == 0) return 0;

        uint256 timeDiff = block.timestamp - s.lastUpdate;
        uint256 yearlyReward = (s.amount * apr) / 100;
        return (yearlyReward * timeDiff) / YEAR;
    }

    function _claimRewards(address user) internal {
        StakeInfo storage s = stakes[user];
        uint256 reward = s.rewardDebt;
        if (reward == 0) return;

        uint256 balance = token.balanceOf(address(this));
        require(balance >= totalStaked + reward, "INSUFFICIENT_REWARDS");

        s.rewardDebt = 0;
        totalRewardsPaid += reward;
        token.safeTransfer(user, reward);
        
        emit RewardsClaimed(user, reward);
    }

    /* ================= VIEW ================= */
    function pendingRewards(address user) public view returns (uint256) {
        StakeInfo storage s = stakes[user];
        return s.rewardDebt + _calculateNewRewards(user);
    }

    function getUserInfo(address user) external view returns (
        uint256 staked,
        uint256 pending,
        uint256 unlockTime,
        uint256 lastUpdate
    ) {
        StakeInfo storage s = stakes[user];
        return (s.amount, pendingRewards(user), s.unlockTime, s.lastUpdate);
    }

    function getContractInfo() external view returns (
        uint256 balance,
        uint256 staked,
        uint256 availableRewards
    ) {
        balance = token.balanceOf(address(this));
        staked = totalStaked;
        availableRewards = balance > staked ? balance - staked : 0;
    }

    /* ================= ADMIN ================= */
    function fundRewards(uint256 amount) external onlyOwner {
        require(amount > 0, "AMOUNT_ZERO");
        token.safeTransferFrom(msg.sender, address(this), amount);
        emit RewardsFunded(amount);
    }

    function setAPR(uint256 newApr) external onlyOwner {
        require(newApr > 0 && newApr <= MAX_APR, "INVALID_APR");
        apr = newApr;
        emit APRUpdated(newApr);
    }

    function emergencyWithdraw(uint256 amount) external onlyOwner {
        uint256 balance = token.balanceOf(address(this));
        uint256 excess = balance > totalStaked ? balance - totalStaked : 0;
        require(excess > 0, "NO_EXCESS");
        require(amount <= excess, "EXCEEDS_EXCESS");

        token.safeTransfer(owner, amount);
        emit EmergencyWithdraw(owner, amount);
    }

    function recoverERC20(address otherToken, uint256 amount) external onlyOwner {
        require(otherToken != address(token), "CANNOT_RECOVER_STAKING_TOKEN");
        IERC20(otherToken).safeTransfer(owner, amount);
        emit ERC20Recovered(otherToken, amount);
    }

    /* ===== MANUAL SYNC ===== */
    function updateMyRewards() external {
        _updateRewards(msg.sender);
    }
}
